<script>
  // Music toggle functionality
  const musicToggle = document.getElementById('music-toggle');
  const backgroundMusic = document.getElementById('background-music');
  
  // Set volume to a pleasant level
  backgroundMusic.volume = 0.3;
  
  // Function to handle audio playback
  function initAudio() {
    // Try to autoplay the music when page loads
    const playPromise = backgroundMusic.play();
    
    if (playPromise !== undefined) {
      playPromise.then(() => {
        // Autoplay succeeded
        musicToggle.checked = true;
      }).catch(error => {
        // Autoplay failed, wait for user interaction
        console.log("Autoplay prevented. Waiting for user interaction.");
        musicToggle.checked = false;
        
        // Enable play on first user interaction
        const enableAudio = () => {
          if (!musicToggle.checked) {
            backgroundMusic.play().then(() => {
              musicToggle.checked = true;
            }).catch(e => {
              console.log("Playback still prevented");
            });
          }
          // Remove listeners after first click
          document.removeEventListener('click', enableAudio);
          document.removeEventListener('keydown', enableAudio);
        };
        
        // Listen for click or keydown events
        document.addEventListener('click', enableAudio);
        document.addEventListener('keydown', enableAudio);
      });
    }
  }
  
  // Initialize audio when page loads
  document.addEventListener('DOMContentLoaded', initAudio);
  
  // Toggle music on checkbox change
  musicToggle.addEventListener('change', function() {
    if (this.checked) {
      backgroundMusic.play().catch(e => {
        console.log("Playback failed:", e);
        this.checked = false;
      });
    } else {
      backgroundMusic.pause();
    }
  });
  
  // Theme toggle functionality with localStorage
  const themeToggle = document.getElementById('theme-toggle');
  
  // Check for saved theme preference
  const savedTheme = localStorage.getItem('theme');
  if (savedTheme === 'dark') {
    themeToggle.checked = true;
  } else if (savedTheme === 'light') {
    themeToggle.checked = false;
  } else if (window.matchMedia && window.matchMedia('(prefers-color-scheme: dark)').matches) {
    // If no saved preference, check system preference
    themeToggle.checked = true;
  }

  // Save theme preference
  themeToggle.addEventListener('change', function() {
    if (this.checked) {
      localStorage.setItem('theme', 'dark');
    } else {
      localStorage.setItem('theme', 'light');
    }
  });
</script>
